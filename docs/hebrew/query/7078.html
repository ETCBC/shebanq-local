Description

Query instruction (mql)

select all objects
where
[clause
  NOTEXIST [phrase function = Objc]
  [phrase FOCUS function = Pred or function = PreS or function = PreC
    NOTEXIST [word lex = 'HLK[']
    NOTEXIST [word lex = 'RDP[']
    NOTEXIST [word lex = '<LH[']
    NOTEXIST [word lex = 'BW>[']
    NOTEXIST [word lex = 'CWB[']
    NOTEXIST [word lex = 'JY>[']
    NOTEXIST [word lex = 'JRD[']
    NOTEXIST [word lex = 'HJH[']
    NOTEXIST [word lex = 'HWH[']
    NOTEXIST [word lex = 'RWY[']
    NOTEXIST [word lex = 'CLX[']
    NOTEXIST [word lex = 'NBV[']
    NOTEXIST [word lex = 'YMX[']
    NOTEXIST [word lex = 'NS<[']
    NOTEXIST [word lex = 'PNH[']
    NOTEXIST [word lex = 'BRX[']
    [word sp = verb And vs <> nif]
  ]
  [phrase FOCUS function IN (Cmpl, Adju, PreC, Loca)
    NOTEXIST [word lex = 'MN']
    NOTEXIST [word lex = '>L']
    NOTEXIST [word lex = 'ZR</']
    NOTEXIST [word lex = 'BN/']
    [word lex = '>XR/']
  ]
]
