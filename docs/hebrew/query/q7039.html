Query instruction (mql)

Select all objects where
[book book IN(Numeri)
  [chapter chapter = 20
    [verse verse >= 1 and verse <= 13
      [word FOCUS sp = verb 
          and not vt = wayq // wayyiqtol
          and not vt = infa // infinitivus absolute
          and not vt = infc // infinitivus constructus
          and not vt = ptca // active participium
          and not vt = ptcp // passive participium
      ]
    ]
  ]
]
